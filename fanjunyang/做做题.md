# EventLoop   
输出打印顺序

``` 
setTimeout(() => {
    console.log(2);
    Promise.resolve().then(() => {
        console.log(6);
    });
}, 0);
Promise.resolve(3).then((data) => {
    console.log(data);          
    return data + 1;
}).then((data) => {
    console.log(data)                
    setTimeout(() => {
        console.log(data + 1)        
        return data + 1;
    }, 1000)
}).then((data) => {
    console.log(data);                
});

// 3 4 undefined 2 6 5
```  

``` 
setTimeout(() => {
    console.log('A');
}, 0);
var obj = {
    func: function () {
        setTimeout(function () {
            console.log('B')
        }, 0);
        return new Promise(function (resolve) {
            console.log('C');
            resolve();
        })
    }
};
obj.func().then(function () {
    console.log('D')
});
console.log('E'); 

// C E D A B
```
